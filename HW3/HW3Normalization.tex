\section*{Нормализация лямбда терма}

\textbf{Исходный терм:}
\[
  ((\lambda a. (\lambda b. b\ b)\ (\lambda b. b\ b))\ b)\ ((\lambda c. (c\ b))\ (\lambda a. a))
\]

begin{align*}
  & ((\lambda a. (\lambda b. b\ b)\ (\lambda b. b\ b))\ b)\ ((\lambda c. (c\ b))\ (\lambda a. a)) \\
  \rightarrow_\beta\ 
  & ((\lambda b. b\ b)\ (\lambda b. b\ b))\ ((\lambda c. (c\ b))\ (\lambda a. a)) 
    && \\
  \rightarrow_\beta\ 
  & \Omega  \ ((\lambda c. (c\ b))\ (\lambda a. a)) 
    && \\
  \rightarrow_\beta\ 
  & \Omega  \ ((\lambda a. a)\ b) 
    &&  \\
  \rightarrow_\beta\ 
  & \Omega  \ b 
    &&
